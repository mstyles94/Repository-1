class Flower: #this is where we set the class
    def __init__(self, name):
        self.name = name

    def grow(self):
        print("The " +self.name + " is growing.") #this is output

    def bloom(self):
        print("The " + self.name + " is blooming.") #this is also output
        
    def die(self):
    	print("The " + self.name + " is dying.") #and a fun one for spooky season
        
def main():
    flower1 = Flower("Rose") #this is an object
    flower1.grow() 
    flower1.bloom()
    flower1.die()
    flower2 = Flower("Daisy")
    flower2.grow()
    flower2.bloom()
    flower2.die()
    flower3 = Flower("Calla Lily")#calla lilies are my favorite flower 
    flower3.grow()
    flower3.bloom()
    flower3.die()
if __name__ == "__main__":
  main()
